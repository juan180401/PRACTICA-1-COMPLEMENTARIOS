import java.util.Scanner;
public class ModuloDeNumero {

    public static void main(String[]ar){
    Scanner teclado = new Scanner(System.in);
    int num1,num2,modu;

    System.out.print("Ingrese segundo número: ");
    num2 = teclado.nextInt();
    modu = num1 % num2;
